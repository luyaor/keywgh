def currency_to_float(s):
    return s if s is np.nan else float(s.strip('$').replace(',', "")) 
def carrier_to_ind(s):
    if s is np.nan: return 0
    return 9 if s == 'Other' else int(s[s.find('_')+1:])
policy_df['quoted_amt'] = policy_df['quoted_amt'].apply(currency_to_float)
policy_df['Prior_carrier_grp'] = policy_df['Prior_carrier_grp'].apply(carrier_to_ind)

def days_between(d1, d2):
    d1 = datetime.strptime(d1, "%Y-%m-%d")
    d2 = datetime.strptime(d2, "%Y-%m-%d")
